37/* Determine the size of an int, if not already specified.
38 * We cannot use sizeof(int), because it is not yet defined
39 * at this stage in the translation of the C program.
40 * Therefore, infer it from UINT_MAX if possible. */
41#ifndef UNITY_INT_WIDTH
42 #ifdef UINT_MAX
43 #if (UINT_MAX == 0xFFFF)
44 #define UNITY_INT_WIDTH (16)
45 #elif (UINT_MAX == 0xFFFFFFFF)
46 #define UNITY_INT_WIDTH (32)
47 #elif (UINT_MAX == 0xFFFFFFFFFFFFFFFF)
48 #define UNITY_INT_WIDTH (64)
49 #endif
50 #else /* Set to default */
51 #define UNITY_INT_WIDTH (32)
52 #endif /* UINT_MAX */
53#endif
54
55/* Determine the size of a long, if not already specified. */
56#ifndef UNITY_LONG_WIDTH
57 #ifdef ULONG_MAX
58 #if (ULONG_MAX == 0xFFFF)
59 #define UNITY_LONG_WIDTH (16)
60 #elif (ULONG_MAX == 0xFFFFFFFF)
61 #define UNITY_LONG_WIDTH (32)
62 #elif (ULONG_MAX == 0xFFFFFFFFFFFFFFFF)
63 #define UNITY_LONG_WIDTH (64)
64 #endif
65 #else /* Set to default */
66 #define UNITY_LONG_WIDTH (32)
67 #endif /* ULONG_MAX */
68#endif
69
70/* Determine the size of a pointer, if not already specified. */
71#ifndef UNITY_POINTER_WIDTH
72 #ifdef UINTPTR_MAX
73 #if (UINTPTR_MAX <= 0xFFFF)
74 #define UNITY_POINTER_WIDTH (16)
75 #elif (UINTPTR_MAX <= 0xFFFFFFFF)
76 #define UNITY_POINTER_WIDTH (32)
77 #elif (UINTPTR_MAX <= 0xFFFFFFFFFFFFFFFF)
78 #define UNITY_POINTER_WIDTH (64)
79 #endif
80 #else /* Set to default */
81 #define UNITY_POINTER_WIDTH UNITY_LONG_WIDTH
82 #endif /* UINTPTR_MAX */
83#endif
