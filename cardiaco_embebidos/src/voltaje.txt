#include <Arduino.h>
#include <stdio.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "driver/adc.h"

#define ADC_VOLTAJE   ADC1_CHANNEL_7  // GPIO35
#define R1 10000.0
#define R2 10000.0

const float CALIBRACION_V = 1.04; 

const float V_MAX = 4.2;  // 100%
const float V_MIN = 3.0;  // 0%

// --- LEER VOLTAJE ---
float leer_voltaje() {
    int adc = adc1_get_raw(ADC_VOLTAJE);
    float Vadc = (adc / 4095.0) * 3.3;
    // Divisor de tensión: (R1+R2)/R2 = 2.0 cuando R1=R2
    float Vreal = Vadc * ((R1 + R2) / R2) * CALIBRACION_V;
    return Vreal;
}

// --- CALCULAR PORCENTAJE ---
float calcular_porcentaje(float voltaje) {
    float porcentaje = (voltaje - V_MIN) / (V_MAX - V_MIN) * 100.0;
    if (porcentaje > 100) porcentaje = 100;
    if (porcentaje < 0) porcentaje = 0;
    return porcentaje;
}

// --- TAREA ---
void tarea_bateria(void *pvParameters) {
    while (1) {
        float sumaVoltaje = 0;
        int numMuestras = 25;

        // 1. FASE DE MUESTREO: Toma 25 muestras, una cada segundo
        for (int i = 0; i < numMuestras; i++) {
            sumaVoltaje += leer_voltaje();
            vTaskDelay(pdMS_TO_TICKS(1000)); // Espera 1 segundo entre muestras
        }

        // 2. CÁLCULO DEL PROMEDIO
        float voltajePromedio = sumaVoltaje / (float)numMuestras;
        float porcentaje = calcular_porcentaje(voltajePromedio);

        // 3. MOSTRAR RESULTADO (Solo ocurre una vez por ciclo)
        printf("--- Reporte Ciclo 65s ---\n");
        printf("V. Promedio (25s): %.2f V | %%: %.2f\n", voltajePromedio, porcentaje);
        printf("-------------------------\n");

        // 4. FASE DE ESPERA: 40 segundos restantes para completar los 65 segundos
        // (25s de muestreo + 40s de espera = 65s totales)
        vTaskDelay(pdMS_TO_TICKS(40000)); 
    }
}

// --- SETUP ---
void setup() {
    Serial.begin(115200);

    adc1_config_width(ADC_WIDTH_BIT_12);
    adc1_config_channel_atten(ADC_VOLTAJE, ADC_ATTEN_DB_11);

    // Creamos la tarea
    xTaskCreate(tarea_bateria, "bateria", 4096, NULL, 5, NULL);
}

// --- LOOP ---
void loop() {
    // El loop principal queda libre para otras funciones
    vTaskDelay(pdMS_TO_TICKS(10000)); 
}
