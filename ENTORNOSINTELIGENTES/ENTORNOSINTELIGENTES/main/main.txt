
/**
 * @file main.c
 * @brief Punto de entrada del sistema de monitoreo biomédico.
 *
 * Hardware conectado:
 *   MAX30100  → SDA=GPIO15, SCL=GPIO16  → I2C_NUM_0  (ritmo cardíaco + SpO2)
 *   MLX90614  → SDA=GPIO18, SCL=GPIO17  → I2C_NUM_1  (temperatura IR)
 *   LED       → GPIO21
 *   BOTÓN     → GPIO33
 *
 * Máquina de estados:
 *
 *   ┌─────────────┐  Botón   ┌──────────────────────┐
 *   │ ESTADO_INICIO│ ──────► │ ESTADO_MONITOREO_RITMO│
 *   │  (espera)    │         │ MAX30100: BPM + SpO2  │
 *   └─────────────┘         └──────────────────────┘
 *          ▲                          │ 2s cumplidos
 *          │  Botón                   ▼
 *          │                ┌──────────────────────┐
 *          └──────────────  │ ESTADO_MONITOREO_TEMP │
 *                           │ MLX90614: Temperatura │
 *                           └──────────────────────┘
 *
 * El botón en cualquier estado regresa a ESTADO_INICIO.
 */

#include <stdio.h>
#include <string.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "esp_log.h"
#include "tareas.h"

static const char *TAG_MAIN = "MAIN";

/* =========================================================================
 * FUNCIONES AUXILIARES LOCALES
 * ========================================================================= */

/** @brief Imprime una línea separadora en el serial para mejor legibilidad. */
static void imprimir_separador(void)
{
    printf("\n------------------------------------------\n");
}

/** @brief Imprime el nombre del estado actual como encabezado en el serial. */
static void imprimir_encabezado_estado(EstadoSistema_t estado)
{
    imprimir_separador();
    switch (estado) {
        case ESTADO_INICIO:
            printf("   ESTADO: INICIO — Esperando botón\n");
            break;
        case ESTADO_MONITOREO_RITMO:
            printf("   ESTADO: MONITOREO DE RITMO CARDÍACO (MAX30100)\n");
            break;
        case ESTADO_MONITOREO_TEMPERATURA:
            printf("   ESTADO: MONITOREO DE TEMPERATURA (MLX90614)\n");
            break;
        default:
            printf("   ESTADO: DESCONOCIDO\n");
            break;
    }
    imprimir_separador();
}

/**
 * @brief Espera a que el botón sea soltado completamente.
 * Previene que una sola pulsación cause múltiples transiciones de estado.
 */
static void esperar_liberacion_boton(void)
{
    while (gpio_get_level(BOTON_GPIO) == 0) {
        vTaskDelay(pdMS_TO_TICKS(10));
    }
    vTaskDelay(pdMS_TO_TICKS(30));  /* Pequeño delay adicional post-liberación */
}

/* =========================================================================
 * FUNCIÓN PRINCIPAL
 * ========================================================================= */

void app_main(void)
{
    /* Banner de inicio en el monitor serial */
    printf("\n\n");
    printf("==========================================\n");
    printf("   SISTEMA DE MONITOREO BIOMÉDICO v2.0   \n");
    printf("   ESP32-S3 | MAX30100 + MLX90614        \n");
    printf("==========================================\n");
    printf("  MAX30100 : GPIO15(SDA) GPIO16(SCL) I2C0\n");
    printf("  MLX90614 : GPIO18(SDA) GPIO17(SCL) I2C1\n");
    printf("  LED      : GPIO21\n");
    printf("  BOTÓN    : GPIO33\n");
    printf("==========================================\n\n");

    /* -----------------------------------------------------------------------
     * INICIALIZACIÓN DE HARDWARE
     * --------------------------------------------------------------------- */
    ESP_LOGI(TAG_MAIN, "Inicializando hardware...");

    /* Bus I2C para el MAX30100 (I2C_NUM_0: GPIO15=SDA, GPIO16=SCL) */
    esp_err_t res_i2c0 = i2c_inicializar_max30100();
    if (res_i2c0 != ESP_OK) {
        ESP_LOGE(TAG_MAIN, "FALLO I2C_NUM_0 (MAX30100). Verificar GPIO15/16.");
    }

    /* Bus I2C para el MLX90614 (I2C_NUM_1: GPIO18=SDA, GPIO17=SCL) */
    esp_err_t res_i2c1 = i2c_inicializar_mlx90614();
    if (res_i2c1 != ESP_OK) {
        ESP_LOGE(TAG_MAIN, "FALLO I2C_NUM_1 (MLX90614). Verificar GPIO18/17.");
    }

    /* Inicializar el MAX30100 (reset + configuración de registros) */
    esp_err_t res_max = max30100_inicializar();
    if (res_max != ESP_OK) {
        ESP_LOGE(TAG_MAIN, "FALLO al inicializar MAX30100. Verificar alimentación.");
    }

    /* LED (GPIO21) y Botón (GPIO33) */
    led_inicializar();
    boton_inicializar();

    ESP_LOGI(TAG_MAIN, "Hardware inicializado.");
    printf("\n[SISTEMA] Presione el botón (GPIO33) para iniciar el monitoreo.\n\n");

    /* -----------------------------------------------------------------------
     * CONTEXTO GLOBAL DEL SISTEMA
     * Contadores de alertas y últimos resultados de cada sensor.
     * --------------------------------------------------------------------- */
    ContextoSistema_t contexto_sistema;
    memset(&contexto_sistema, 0, sizeof(contexto_sistema));

    /* -----------------------------------------------------------------------
     * VARIABLES DE CONTROL DE LA MÁQUINA DE ESTADOS
     * --------------------------------------------------------------------- */
    EstadoSistema_t estado_actual = ESTADO_INICIO;

    /*
     * Flags de modo alerta para la SIGUIENTE ventana de monitoreo.
     * Cuando se activa una alerta, el siguiente ciclo del mismo sensor
     * tomará más muestras (modo alerta = true).
     */
    bool modo_alerta_siguiente_ritmo       = false;
    bool modo_alerta_siguiente_temperatura = false;

    /* Flag para evitar imprimir el encabezado de INICIO en cada iteración */
    bool encabezado_inicio_impreso = false;

    /* -----------------------------------------------------------------------
     * BUCLE PRINCIPAL — MÁQUINA DE ESTADOS
     * --------------------------------------------------------------------- */
    while (1) {

        /* =================================================================
         * ESTADO 1: INICIO
         * El sistema espera inactivo hasta que el usuario presione el botón.
         * El LED está apagado. No se realiza ninguna medición.
         * ================================================================= */
        if (estado_actual == ESTADO_INICIO) {

            if (!encabezado_inicio_impreso) {
                imprimir_encabezado_estado(ESTADO_INICIO);
                led_apagar();
                encabezado_inicio_impreso = true;
            }

            if (boton_esta_presionado()) {
                printf("[SISTEMA] Botón presionado → Iniciando monitoreo\n");
                esperar_liberacion_boton();

                /* Reiniciar contadores y flags al comenzar sesión nueva */
                memset(&contexto_sistema, 0, sizeof(contexto_sistema));
                modo_alerta_siguiente_ritmo       = false;
                modo_alerta_siguiente_temperatura = false;
                encabezado_inicio_impreso         = false;

                printf("[SISTEMA] Nueva sesión iniciada. Contadores reiniciados.\n");
                estado_actual = ESTADO_MONITOREO_RITMO;
            }

            vTaskDelay(pdMS_TO_TICKS(50));  /* Polling del botón cada 50 ms */
        }

        /* =================================================================
         * ESTADO 2: MONITOREO DE RITMO CARDÍACO
         * Adquiere muestras del MAX30100 durante la ventana configurada,
         * calcula BPM y SpO2, y evalúa si hay alerta roja.
         * ================================================================= */
        else if (estado_actual == ESTADO_MONITOREO_RITMO) {

            imprimir_encabezado_estado(ESTADO_MONITOREO_RITMO);
            led_encender();  /* LED encendido indica que el sistema está midiendo */

            /* Ejecutar tarea de monitoreo de ritmo */
            tarea_monitorear_ritmo(modo_alerta_siguiente_ritmo,
                                   &contexto_sistema.ultimo_ritmo);

            /* Verificar botón después de la ventana de monitoreo */
            if (boton_esta_presionado()) {
                printf("[SISTEMA] Botón presionado → Regresando a INICIO\n");
                esperar_liberacion_boton();
                led_apagar();
                estado_actual             = ESTADO_INICIO;
                encabezado_inicio_impreso = false;
                continue;
            }

            /* Evaluar alerta roja y actualizar flag para siguiente ciclo */
            bool alerta_roja = evaluar_alerta_roja(
                &contexto_sistema.ultimo_ritmo, &contexto_sistema);

            contexto_sistema.ultimo_ritmo.alerta_activa = alerta_roja;
            modo_alerta_siguiente_ritmo = alerta_roja;

            /* Si no hubo alerta, resetear el modo normal para el siguiente ciclo */
            if (!alerta_roja) {
                modo_alerta_siguiente_ritmo = false;
            }

            /* Verificar si el acumulado de alertas indica crisis */
            verificar_posible_crisis(&contexto_sistema);

            /* Transición al estado de temperatura */
            led_apagar();
            estado_actual = ESTADO_MONITOREO_TEMPERATURA;
        }

        /* =================================================================
         * ESTADO 3: MONITOREO DE TEMPERATURA
         * Adquiere lecturas del MLX90614 durante la ventana configurada,
         * calcula el promedio de temperatura y evalúa si hay alerta amarilla.
         * ================================================================= */
        else if (estado_actual == ESTADO_MONITOREO_TEMPERATURA) {

            imprimir_encabezado_estado(ESTADO_MONITOREO_TEMPERATURA);
            led_encender();

            /* Ejecutar tarea de monitoreo de temperatura */
            tarea_monitorear_temperatura(modo_alerta_siguiente_temperatura,
                                          &contexto_sistema.ultima_temperatura);

            /* Verificar botón después de la ventana de monitoreo */
            if (boton_esta_presionado()) {
                printf("[SISTEMA] Botón presionado → Regresando a INICIO\n");
                esperar_liberacion_boton();
                led_apagar();
                estado_actual             = ESTADO_INICIO;
                encabezado_inicio_impreso = false;
                continue;
            }

            /* Evaluar alerta amarilla y actualizar flag para siguiente ciclo */
            bool alerta_amarilla = evaluar_alerta_amarilla(
                &contexto_sistema.ultima_temperatura, &contexto_sistema);

            contexto_sistema.ultima_temperatura.alerta_activa = alerta_amarilla;
            modo_alerta_siguiente_temperatura = alerta_amarilla;

            if (!alerta_amarilla) {
                modo_alerta_siguiente_temperatura = false;
            }

            /* Verificar si el acumulado de alertas indica crisis */
            verificar_posible_crisis(&contexto_sistema);

            /* Transición de regreso al monitoreo de ritmo */
            led_apagar();
            estado_actual = ESTADO_MONITOREO_RITMO;
        }

        /* Estado desconocido: resetear (programación defensiva) */
        else {
            ESP_LOGW(TAG_MAIN, "Estado inválido detectado. Reseteando a INICIO.");
            estado_actual             = ESTADO_INICIO;
            encabezado_inicio_impreso = false;
        }

    } /* while(1) */
}
